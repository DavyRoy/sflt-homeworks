#!/bin/bash


USER_HOME="/home/admin"


BACKUP_DIR="/tmp/backup"


LOG_FILE="/home/admin/backup.log"


rsync -avh --delete --checksum --exclude='.*' "$USER_HOME/" "$BACKUP_DIR" >> "$LOG_FILE" 2>&1


if [ $? -eq 0 ]; then
    echo "$(date) - Резервное копирование успешно завершено" >> "$LOG_FILE"
